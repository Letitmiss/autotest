# 切片
  截取列表

```
#切片
L = ['Michael', 'Sarah', 'Tracy', 'Bob', 'Jack']
#取前三个元素,辩证一个新的list
L2 = []

# 循环方式
for i in range(3):
    L2.append(L[i])
print(L2)

# Python提供了切片（Slice）操作符，能大大简化这种操作
print(L[0:3])
# 表示索引从0开始,不包括3 [0,3) 第一个索引是0可以省略

#也可以从索引1开始，取出2个元素出来
print(L[1:3])

# 倒取切片 ,取出倒数两个
print(L[-2:])

#取出倒数第三个
print(L[-3:-2])

L3=list(range(100))
# 前10个
print(L3[:10])
# 复制
L4=L3[:]
print(L4)

#tuple也可以切片
t = (1,2,3,4,5)
print(t[:4])

# 字符串也可以切片
str = 'iloveyou'
print(str[:6])
```
